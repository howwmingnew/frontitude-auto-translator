**Step 5: Create js/app.js** (~80-120 lines)
This is the LAST script to load. It wires everything together:

```javascript
(function () {
  'use strict';
  var App = window.App;

  function init() {
    // Theme
    App.applyTheme(App.getPreferredTheme());

    // UI language
    var savedLang = localStorage.getItem('translate_ui_lang');
    if (savedLang) {
      App.setState({ uiLang: savedLang });
    } else {
      var browserLang = (navigator.language || '').toLowerCase();
      if (browserLang.startsWith('zh')) App.setState({ uiLang: 'zh-TW' });
      else if (browserLang.startsWith('ko')) App.setState({ uiLang: 'ko' });
      else App.setState({ uiLang: 'en' });
    }
    App.applyI18n();

    // Context prompt
    var savedPrompt = localStorage.getItem('translate_context_prompt');
    if (savedPrompt !== null) {
      App.dom.contextPrompt.value = savedPrompt;
      App.setState({ contextPrompt: savedPrompt });
    } else {
      var defaultVal = App.t('defaultContextPrompt');
      App.dom.contextPrompt.value = defaultVal;
      App.setState({ contextPrompt: defaultVal });
    }

    // Provider
    App.initProvider();

    // Sidebar / drawer
    App.toggleSidebarVisibility();

    // Collapsible sections
    App.initCollapsible(App.dom.providerCollapsible, App.dom.providerCollapsibleBody, 'provider');
    App.initCollapsible(App.dom.langCollapsible, App.dom.langCollapsibleBody, 'langs');

    // Event listeners
    // Theme toggle
    App.dom.themeToggle.addEventListener('click', function () {
      var s = App.getState();
      var newTheme = s.theme === 'dark' ? 'light' : 'dark';
      App.applyTheme(newTheme);
      App.setState({ theme: newTheme });
      localStorage.setItem('translate_theme', newTheme);
    });

    // Language switcher
    // ... wire up lang-switcher buttons ...

    // Translate button
    App.dom.translateBtn.addEventListener('click', function () {
      App.showTranslateConfirm();
    });

    // Download button
    App.dom.downloadBtn.addEventListener('click', function () {
      App.downloadLanguageFile();
    });

    // Context prompt change
    App.dom.contextPrompt.addEventListener('input', function () {
      App.setState({ contextPrompt: this.value });
      localStorage.setItem('translate_context_prompt', this.value);
    });

    // Window resize for sidebar
    window.addEventListener('resize', App.toggleSidebarVisibility);
  }

  App.init = init;
  init();
})();
```
